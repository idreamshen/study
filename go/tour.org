#+TITLE: A Tour of Go

#+PROPERTY: header-args :exports both

* Packages
  #+begin_src go
    package main

    import (
        "fmt"
        "math/rand"
    )

    func main() {
        fmt.Println("My favorite number is", rand.Intn(10))
    }
  #+end_src

#+RESULTS:
: My favorite number is 1

* Imports
  #+begin_src go
    package main

    import (
        "fmt"
        "math"
    )

    func main() {
        fmt.Printf("Now you have %g problems.\n", math.Sqrt(7))
    }
  #+end_src

  #+RESULTS:
  : Now you have 2.6457513110645907 problems.

* Exported names
  #+begin_src go
    package main

    import (
        "fmt"
        "math"
    )

    func main() {
        fmt.Println(math.Pi)
    }
  #+end_src

  #+RESULTS:
  : 3.141592653589793
